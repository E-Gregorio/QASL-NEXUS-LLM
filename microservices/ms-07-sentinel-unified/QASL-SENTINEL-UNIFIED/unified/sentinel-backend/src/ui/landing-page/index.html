Quick Start

Comienza en minutos

Instalación simple con npm o Docker

# Instalar globalmente
npm install -g qasl-api-sentinel

# O clonar el repositorio
git clone https://github.com/qasl-api-sentinel/sentinel
cd sentinel
npm install

# Iniciar el servidor
npm run server

# En otra terminal, iniciar el monitoreo
sentinel start --ai-insights
# Usando Docker Compose
docker-compose up -d

# O Docker directamente
docker run -d \
  --name sentinel \
  -p 3000:3000 \
  -v ./config:/app/config \
  qasl/api-sentinel:latest

# Ver logs
docker logs -f sentinel
// sentinel.config.js
export default {
  server: {
    host: 'localhost',
    port: 3000
  },
  monitoring: {
    interval: 30000,
    timeout: 10000,
    retries: 3
  },
  ml: {
    anomalyDetection: true,
    failurePrediction: true
  },
  webhooks: {
    slack: process.env.SLACK_WEBHOOK
  }
}
